205  static void LoadIniFile(const std::filesystem::path &path)
206  {
207  boost::property_tree::ptree pt;
208  try {
209  boost::property_tree::ini_parser::read_ini(path.generic_string(), pt);
210  } catch (std::exception &e) {
211  std::cout << e.what() << std::endl;
212  }
213 
214  // loop through the sections
215  for (auto &section : pt) {
216  for (auto &value : section.second) {
217  std::vector<std::string> paths;
218  // parse the section.second with slash and put it in the vector
219  while (value.second.data().find('/') != std::string::npos) {
220  paths.push_back(value.second.data().substr(0, value.second.data().find('/')));
221  value.second.data().erase(0, value.second.data().find('/') + 1);
222  }
223  if (section.first == "keybind") {
224  loadKeybind(value.first, value.second.data());
225  continue;
226  }
227  paths.push_back(value.second.data());
228 
229  if (section.first == "texture")
230  LoadTexture(value.first, paths);
231  if (section.first == "bgm")
232  LoadBGM(value.first, paths);
233  if (section.first == "sfx")
234  LoadSFX(value.first, paths);
235  if (section.first == "font")
236  LoadFont(value.first, paths);
237  }
238  }
